```css
/* 设置文字间距 */
letter-spacing: normal;

/* 设置字体样式(斜体,正体,下划线) */
font-style: normal;

/* 换行 */
white-space: normal;
word-break: break-word;
word-wrap: break-word;

/* 下划线,上划线,删除线 */
text-decoration: none;
text-decoration: underline;

/* 文本溢出 使用省略号 */
text-overflow: ellipsis;

/* 文档高度 */
line-height: 1.3;

/* 序列 不显示序列符号 */
list-style: none;

/* 不支持鼠标选中 */
user-select: none;


/* 最小 767像素 */
@media (max-width: 767px){}



```
